  const el = document.getElementById(containerId);
  if (!el) return;
  cities.forEach((c, i) => {

function applyChartFilters() {
  const activeBtns = [...document.querySelectorAll('#cityFilters .city-btn.active')].map(b=>b.textContent);
  filteredCities = activeBtns.length ? cities.filter(c=>activeBtns.includes(c.name)) : cities;
  updateMainChart();

function updateMainChart() {
  if (!charts.aqi) return;
  const labels = filteredCities.map(c=>c.name);
  let data, label;
  if (activePollutant === 'aqi') { data = filteredCities.map(c=>c.aqi); label = 'AQI'; }
  else if (activePollutant === 'pm25') { data = filteredCities.map(c=>c.pm25); label = 'PM2.5 (µg/m³)'; }
  else if (activePollutant === 'pm10') { data = filteredCities.map(c=>c.pm10); label = 'PM10 (µg/m³)'; }
  else if (activePollutant === 'no2') { data = filteredCities.map(c=>c.no2); label = 'NO₂ (ppb)'; }
  const colors = filteredCities.map(c=>c.color);
  charts.aqi.data.labels = labels;
  charts.aqi.data.datasets[0].data = data;
  charts.aqi.data.datasets[0].label = label;
  charts.aqi.data.datasets[0].backgroundColor = colors.map(c=>c+'66');
  charts.aqi.data.datasets[0].borderColor = colors;
  charts.aqi.update();
}

function setPollutant(p, btn) {

  updateMainChart();
}

        : `Moderate AQI can cause occasional nighttime coughing in sensitive individuals. Keep windows closed at night to reduce indoor PM2.5 accumulation.`
    )}
    ${healthItem('⚡','Energy Levels',airRisk,
      `At AQI ${c.aqi}, the body uses extra resources combating oxidative stress from particulates. This translates to 10-25% reduction in perceived energy, brain fog, and afternoon fatigue cycles.`
    )}
    ${healthItem('🏃','Exercise Performance',isPoor ? 'high' : 'mod',
      isPoor
        ? `Outdoor exercise at AQI ${c.aqi} is NOT recommended. Breathing rate doubles during exercise, increasing PM2.5 inhalation by 10×. Use indoor gyms with HEPA filtration.`
        : `Light to moderate outdoor exercise is acceptable. Avoid peak traffic hours (7-9am, 5-8pm). Morning (before 8am) is the relatively better window.`
    )}
    ${healthItem('🧠','Long-Term Health Risk',airRisk,
      `Each 10µg/m³ increase in PM2.5 is associated with 8% higher cardiovascular mortality risk. ${c.name}'s annual exposure level is approximately ${(c.pm25 * 14).toFixed(0)}µg/m³ above WHO annual guideline (5µg/m³).`
    )}

    <div class="section-header" style="margin-top:20px;"><h2>💧 Water Quality Health Impact</h2><div class="section-line"></div></div>
    <div style="margin-bottom:6px;"><span class="aqi-pill ${c.waterScore<40?'aqi-poor':c.waterScore<60?'aqi-moderate':'aqi-good'}">${c.wqi} Water Quality</span></div>

    ${healthItem('💇','Hair Fall',waterRisk,
      c.waterScore < 40
        ? `${c.name}'s water has high TDS, heavy metals (Pb, Fe, Ca), and microbial contamination. Hard water deposits calcium/magnesium on scalp, blocking follicles and causing increased hair fall. Risk: HIGH.`
        : `Moderate hardness may cause some follicle buildup. Use chelating shampoo weekly to remove mineral deposits.`
    )}
    ${healthItem('🌵','Hair Dryness',waterRisk,
      `Hard water (${c.waterHardness} hardness detected) strips natural oils from hair shaft. Calcium ions bind to hair protein, making it brittle and dry. Regular deep conditioning is essential.`
    )}
    ${healthItem('🧴','Scalp Health','mod',
      c.waterScore < 40
        ? `Bacterial and fungal growth risk elevated with contaminated water. Chlorine in treated water kills beneficial scalp microbiome. May worsen dandruff, seborrheic dermatitis.`
        : `Use filtered water for final rinse. Anti-fungal shampoo bi-weekly recommended.`
    )}
    ${healthItem('✨','Skin Dryness',waterRisk,
      `Hard water reacts with soap to form scum, leaving residue on skin. This clogs pores and disrupts skin barrier. Dryness, itchiness, and eczema flares are common in ${c.name}.`
    )}
    ${healthItem('😤','Acne & Breakouts','mod',
      `Mineral deposits and microbial load in water contribute to comedonal acne. Chlorine residue disrupts skin microbiome. Double-cleansing with micellar water (not tap water) recommended.`
    )}
    ${healthItem('🌸','Sensitive Skin',waterRisk,
      c.waterScore < 40
        ? `Very high risk for sensitive skin types. Heavy metals (even trace Pb, As) cause contact dermatitis. Use only filtered/bottled water for face washing if skin is reactive.`
        : `Use lukewarm filtered water. Avoid prolonged contact with tap water on skin.`
    )}
  `;
}

function healthItem(icon, title, risk, desc) {
  const riskMap = {
    low: { cls:'risk-low', label:'🟢 Low Risk' },
    mod: { cls:'risk-mod', label:'🟡 Moderate Risk' },
    high: { cls:'risk-high', label:'🔴 High Risk' }
  };
  const r = riskMap[risk] || riskMap['mod'];
  return `
    <div class="health-item">
      <div class="health-icon">${icon}</div>
      <div>
        <div class="risk-badge ${r.cls}">${r.label}</div>
        <div class="health-title">${title}</div>
        <div class="health-desc">${desc}</div>
      </div>
    </div>`;
}
