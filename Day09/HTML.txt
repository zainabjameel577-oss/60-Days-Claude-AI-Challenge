// ══════════════════════════════════════════════
//  TARGETS  (Mifflin–St Jeor + ICMR guidelines)
// ══════════════════════════════════════════════
function calcTargets() {
  const age    = +document.getElementById('age').value    || 28;
  const gender = document.getElementById('gender').value;
  const h      = +document.getElementById('height').value || 170;
  const w      = +document.getElementById('weight').value || 70;
  const af     = +document.getElementById('activity').value || 1.55;

  let bmr = gender === 'male'
    ? 10*w + 6.25*h - 5*age + 5
    : 10*w + 6.25*h - 5*age - 161;

  const kcal = Math.round(bmr * af);
  const protein = Math.round(w * 1.6);        // 1.6g/kg moderate activity
  const fat    = Math.round(kcal * 0.27 / 9);
  const carbs  = Math.round((kcal - protein*4 - fat*9) / 4);
  const fiber  = gender === 'male' ? 38 : 25;

  // Micronutrient targets (ICMR / RDA)
  const iron    = gender === 'female' && age < 50 ? 29 : 17;
  const calcium = age < 50 ? 1000 : 1200;
  const vitC    = 40;
  const vitD    = 15;
  const vitB12  = 2.2;

  targets = { kcal, protein, carbs, fat, fiber, iron, calcium, vitC, vitD, vitB12 };

  // Update sidebar targets display
  document.getElementById('t-kcal').textContent    = kcal + ' kcal';
  document.getElementById('t-protein').textContent = protein + ' g';
  document.getElementById('t-carbs').textContent   = carbs + ' g';
  document.getElementById('t-fat').textContent     = fat + ' g';
  document.getElementById('t-fiber').textContent   = fiber + ' g';
}

// ══════════════════════════════════════════════
//  TOTALS
// ══════════════════════════════════════════════
function calcTotals() {
  const keys = ['kcal','protein','carbs','fat','fiber','iron','calcium','vitC','vitD','vitB12'];
  const totals = {};
  keys.forEach(k => totals[k] = 0);

  foodLog.forEach(entry => {
    const d = FOOD_DB[entry.name];
    const factor = entry.grams / 100;
    keys.forEach(k => {
      totals[k] += (d[k] || 0) * factor;
    });
  });

  // Round
  keys.forEach(k => totals[k] = Math.round(totals[k] * 10) / 10);
  return totals;
}

  renderSessionTotals(totals);
  renderRecommendations(totals);
}

  });

  const excs = scored.filter(n => n.pct > 120).sort((a,b) => b.pct - a.pct).slice(0,5);

  const defEl = document.getElementById('deficiency-list');
  const excEl = document.getElementById('excess-list');

      <div style="font-size:12px;color:var(--danger);font-weight:600">${Math.round(n.pct)}%</div>

function addFood() {
  const name = document.getElementById('food-select').value;
  const qty  = +document.getElementById('food-qty').value;
  const unit = document.getElementById('food-unit').value;
  if (!name) { showToast('Please select a food item'); return; }
  if (!qty || qty <= 0) { showToast('Please enter a valid quantity'); return; }

  const multiplier = UNIT_MULTIPLIERS[unit] || 1;
  const grams = qty * multiplier;
  const id = Date.now();

  foodLog.push({ id, name, qty, unit, grams });
  renderLogTable();
  recalc();
  showToast(`Added ${name} (${qty} ${unit})`);
}

function removeFood(id) {
  foodLog = foodLog.filter(e => e.id !== id);
  renderLogTable();
  recalc();
}

function updateQty(id, newQty) {
  const entry = foodLog.find(e => e.id === id);
  if (!entry) return;
  const qty = +newQty;
  if (!qty || qty <= 0) return;
  entry.qty = qty;
  const multiplier = UNIT_MULTIPLIERS[entry.unit] || 1;
  entry.grams = qty * multiplier;
  recalc();
}

function renderLogTable() {
  const tbody = document.getElementById('log-tbody');
  if (!foodLog.length) {
    tbody.innerHTML = '<tr><td colspan="8"><div class="empty-state"><div class="icon">🍽️</div><p>No food logged yet. Add your first entry above.</p></div></td></tr>';
    return;
  }

  tbody.innerHTML = foodLog.map(entry => {
    const d = FOOD_DB[entry.name];
    const f = entry.grams / 100;
    return `<tr>
      <td style="font-weight:500">${entry.name}</td>
      <td><input class="editable" type="number" value="${entry.qty}" min="1"
        onchange="updateQty(${entry.id}, this.value)" onblur="updateQty(${entry.id}, this.value)"></td>
      <td style="color:var(--text-muted)">${entry.unit}</td>
      <td>${Math.round(d.kcal*f)}</td>
      <td>${Math.round(d.protein*f*10)/10}</td>
      <td>${Math.round(d.carbs*f*10)/10}</td>
      <td>${Math.round(d.fat*f*10)/10}</td>
      <td><button class="btn btn-danger btn-sm" onclick="removeFood(${entry.id})">✕</button></td>
    </tr>`;
  }).join('');
}

  const t = targets;

    if (pref === 'veg') {
      recs.push({ type:'add', icon:'🧀', title:'Add Paneer or Dal', desc:'Your protein intake is significantly below target. Add 100g paneer (18g protein) or a bowl of dal (9g protein) to your meals.' });
    } else if (pref === 'egg') {
      recs.push({ type:'add', icon:'🥚', title:'Add Eggs to your meal', desc:`You're ${Math.round(t.protein - totals.protein)}g short on protein. 2 whole eggs add ~26g protein and support muscle health.` });
    } else {
      recs.push({ type:'add', icon:'🍗', title:'Add Chicken or Fish', desc:`You need ${Math.round(t.protein - totals.protein)}g more protein. 100g chicken breast provides 31g protein. Fish is also an excellent option.` });
    }
  }

  if (totals.iron < t.iron * 0.6) {
    if (pref === 'veg') {
      recs.push({ type:'add', icon:'🥬', title:'Include Spinach', desc:'Your iron levels are low. Spinach has 2.7mg iron per 100g. Pair with vitamin C sources (potato, lemon) to improve absorption.' });
    } else {
      recs.push({ type:'add', icon:'🐟', title:'Add Fish or Spinach', desc:`Iron is at ${Math.round(totals.iron/t.iron*100)}% of target. Fish and spinach are excellent sources. Avoid tea/coffee with iron-rich foods.` });
    }
  }

  if (totals.calcium < t.calcium * 0.6) {
    recs.push({ type:'add', icon:'🥛', title:'Boost Calcium Intake', desc:`Your calcium is at ${Math.round(totals.calcium/t.calcium*100)}% of target. Add a glass of milk (113mg/100ml) or curd (121mg/100g) to each meal.` });
  }

  if (totals.vitC < t.vitC * 0.5) {
    recs.push({ type:'add', icon:'🍊', title:'Add Vitamin C sources', desc:'Vitamin C is low. Add spinach, potato, or banana to your diet. Spinach has 28mg/100g and potato has 19.7mg/100g of Vitamin C.' });
  }

  if (pref !== 'veg' && totals.vitD < t.vitD * 0.5) {
    recs.push({ type:'add', icon:'☀️', title:'Vitamin D Boost', desc:'Vitamin D is critically low. Fish (especially fatty fish) has 440mcg/100g. Eggs also provide 87mcg/100g. Sunlight exposure also helps.' });
  }

  if (totals.vitB12 < t.vitB12 * 0.5 && pref === 'veg') {
    recs.push({ type:'add', icon:'🥛', title:'Vitamin B12 Alert', desc:'B12 is very low — a common concern for vegetarians. Include milk, curd, and paneer daily. Consider a B12 supplement.' });
  }

  if (totals.fiber < t.fiber * 0.5) {
    recs.push({ type:'add', icon:'🫘', title:'Add Fiber-Rich Foods', desc:'Fiber is below 50% of target. Add oats (10.6g/100g), chana (7.6g/100g), or rajma (6.4g/100g) to improve gut health.' });
  }

  // --- SWAP recommendations ---
  const loggedNames = foodLog.map(e => e.name);

  if (loggedNames.includes('Bread') && pref === 'veg') {
    recs.push({ type:'swap', icon:'🔄', title:'Swap Bread → Roti', desc:'Roti has more fiber (2.7g vs 2.7g similar) but fewer additives and preservatives. It also has a lower glycemic impact.' });
  }

  if (loggedNames.includes('Rice') && totals.carbs > t.carbs * 1.2) {
    recs.push({ type:'swap', icon:'🔄', title:'Swap White Rice → Oats', desc:'You are over your carbs target. Replacing rice with oats reduces carbs, adds fiber (10.6g), and keeps you full longer.' });
  }

  if (loggedNames.includes('Potato') && totals.carbs > t.carbs * 1.1) {
    recs.push({ type:'swap', icon:'🔄', title:'Swap Potato → Spinach', desc:'Potato is high in carbs (17g/100g). Spinach has only 3.6g carbs and delivers much more iron, calcium, and Vitamin C.' });
  }

  // --- PORTION recommendations ---
  if (totals.fat > t.fat * 1.3) {
    const highFat = foodLog.filter(e => FOOD_DB[e.name].fat > 10).map(e => e.name);
    recs.push({ type:'portion', icon:'⚖️', title:'Reduce Fat Portions', desc:`Fat intake is ${Math.round(totals.fat/t.fat*100)}% of target. Consider smaller portions of ${highFat.join(', ') || 'high-fat foods'}.` });
  }

  if (totals.kcal > t.kcal * 1.15) {
    recs.push({ type:'portion', icon:'⚖️', title:'Calorie Surplus', desc:`You are ${Math.round(totals.kcal - t.kcal)} kcal over your daily target. Consider smaller portions or choose lower-calorie foods for remaining meals.` });
  }

  if (totals.carbs > t.carbs * 1.25) {
    recs.push({ type:'portion', icon:'⚖️', title:'Reduce Carbohydrate Load', desc:`Carbs at ${Math.round(totals.carbs/t.carbs*100)}% of target. Reduce rice or roti portions and supplement with protein-rich foods.` });
  }

  if (!recs.length) {
    recs.push({ type:'add', icon:'🎯', title:'Great Balance!', desc:`Your diet looks well-balanced. Keep maintaining this variety and ensure you hit your ${t.kcal} kcal target.` });
  }

  const tagMap = { add:'tag-add', swap:'tag-swap', portion:'tag-portion' };
  const tagLabel = { add:'Add Food', swap:'Food Swap', portion:'Portion Adjustment' };

  container.innerHTML = recs.map(r => `
    <div class="rec-item">
      <span class="rec-icon">${r.icon}</span>
      <div>
        <span class="rec-tag ${tagMap[r.type]}">${tagLabel[r.type]}</span>
        <div class="rec-title">${r.title}</div>
        <div class="rec-desc">${r.desc}</div>
      </div>
    </div>
  `).join('');
}

function switchPage(name) {
  document.querySelectorAll('.page').forEach(p => p.classList.remove('active'));
  document.querySelectorAll('.nav-tab').forEach(t => t.classList.remove('active'));
  document.getElementById('page-' + name).classList.add('active');
  event.target.classList.add('active');
  if (name === 'dashboard') {
    const totals = calcTotals();
    renderMacroChart(totals);
  }
}
